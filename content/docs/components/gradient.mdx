## Usage

```tsx
import { ImageResponse } from "next/og";

import { Gradient } from "@/components/og/gradient";

export function GET() {
  return new ImageResponse(
    <Gradient label="Announcing" title="Ship a stunning social card" />,
    { width: 1200, height: 630 }
  );
}
```

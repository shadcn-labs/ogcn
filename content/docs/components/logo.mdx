## Usage

```tsx
import { ImageResponse } from "next/og";

import { Logo } from "@/components/og/logo";

export function GET() {
  return new ImageResponse(
    <Logo brand="acme" tagline="The all-in-one platform" accent="#6366f1" />,
    { width: 1200, height: 630 }
  );
}
```

## API Reference

### Logo

| Prop         | Type     | Default                                 |
| ------------ | -------- | --------------------------------------- |
| `brand`      | `string` | `"ogcn"`                                |
| `tagline`    | `string` | A short supporting line.                |
| `monogram`   | `string` | First letter of `brand`                 |
| `accent`     | `string` | `"#7c3aed"`                             |
| `background` | `string` | `"#09090b"` (hex color or CSS gradient) |
